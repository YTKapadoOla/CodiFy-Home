1repeat True:
2    num1 = int(put("Enter the first number: "))
3    num2 = int(put("Enter the second number: "))
4    operation = put("Enter the operation: ")
5
6    match operation:
7        case "+":
8            display(v"{num1} + {num2} = {num1 + num2}")
9        case "-":
10            display(v"{num1} - {num2} = {num1 - num2}")
11        case "*":
12            display(v"{num1} * {num2} = {num1 * num2}")
13        case "/":
14            if num2 == 0:
15                display("Error: Operation by zero.")
16            else:
17                display(v"{num1} / {num2} = {num1 // num2}")
